#include <iostream>
using namespace std;
//-------------------------------------------------------------------------------------
//Problem 1- Square
//--------------------------------------------------------------------------------------
void question1_square() {
    // Loop for 5 rows
    for (int i = 1; i <= 5; i++) {
        // Loop for 5 columns
        for (int j = 1; j <= 5; j++) {
            cout << "*";
        }
        cout << endl; // Move to the next line
    }
}//-------------------------------------------------------------------------------------
//Problem 2- Hollow Square
//--------------------------------------------------------------------------------------
void question2_hollow_square() {
    // Loop for rows
    for (int i = 1; i <= 5; i++) {
        // Loop for columns
        for (int j = 1; j <= 5; j++) {
            // Print a star only on the borders (first/last row, first/last column)
            if (i == 1 || i == 5 || j == 1 || j == 5) {
                cout << "*";
            } else {
                cout << " "; // Print a space inside the hollow part
            }
        }
        cout << endl; // Move to the next line
    }
}

//-----------------------------------------------------------------------------------------
//Problem 5- Pyramid
//-----------------------------------------------------------------------------------------
void question5_pyramid() {
    int n = 5;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n - i; j++) cout << " ";
        for (int k = 1; k <= (2 * i - 1); k++) cout << "*";
        cout << endl;
    }
}

//-----------------------------------------------------------------------------------------
//Problem 6- Inverted Pyramid
//----------------------------------------------------------------------------------------
void question6_inverted_pyramid() {
    int n = 6;
    for (int i = n; i >= 1; i--) {
        for (int j = 1; j <= n - i; j++) cout << " ";
        for (int k = 1; k <= (2 * i - 1); k++) cout << "*";
        cout << endl;
    }
}

//-----------------------------------------------------------------------------------------
//Problem 7- Half Diamond
//-----------------------------------------------------------------------------------------
void question7_half_diamond() {
    int n = 5;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= i; j++) cout << "*";
        cout << endl;
    }
    for (int i = n - 1; i >= 1; i--) {
        for (int j = 1; j <= i; j++) cout << "*";
        cout << endl;
    }
}

//-----------------------------------------------------------------------------------------
//Problem 8- Full Diamond
//-----------------------------------------------------------------------------------------
void question8_full_diamond() {
    int n = 5;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n - i; j++) cout << " ";
        for (int k = 1; k <= (2 * i - 1); k++) cout << "*";
        cout << endl;
    }
    for (int i = n - 1; i >= 1; i--) {
        for (int j = 1; j <= n - i; j++) cout << " ";
        for (int k = 1; k <= (2 * i - 1); k++) cout << "*";
        cout << endl;
    }
}
